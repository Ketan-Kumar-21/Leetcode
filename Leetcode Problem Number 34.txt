class Solution:
    def searchRange(self, nums: List[int], target: int) -> List[int]:
        def first(nums, target):
            first = -1
            low = 0
            high = len(nums) - 1
            while low <= high:
                mid = (low + high) // 2
                if nums[mid] == target:
                    first = mid
                    high = mid - 1  
                elif target < nums[mid]:
                    high = mid - 1
                else:
                    low = mid + 1
            return first

        def last(nums, target):
            last = -1
            low = 0
            high = len(nums) - 1
            while low <= high:
                mid = (low + high) // 2
                if nums[mid] == target:
                    last = mid
                    low = mid + 1  
                elif target < nums[mid]:
                    high = mid - 1
                else:
                    low = mid + 1
            return last

        return [first(nums, target), last(nums, target)]



TC: O(logn)+O(logn)
